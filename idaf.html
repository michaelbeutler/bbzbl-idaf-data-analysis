# Helpers
def format_hist(hist):
    for ax in hist.flatten():
        ax.set_xlabel("in %")
        ax.set_ylabel("Häufigkeit")


def format_boxplot(boxplot, title):
    plt.title(title)
    plt.suptitle('')
    boxplot.set_xlabel("")
    boxplot.set_ylabel("Häufigkeit")


def save_diagram(diagram, filename):

def save_hist(hist, name):
    save_diagram(
        hist, './diagrams/{name}hist.png'.format(name=name))


def save_boxplot(hist, name):
    save_diagram(
        hist, './diagrams/{name}boxplot.png'.format(name=name))
